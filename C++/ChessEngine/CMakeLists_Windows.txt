project(ChessEngineProject)

set(CMAKE_CXX_STANDARD_REQUIRED ON)

        ../Common/FileUtils/FileUtils.cpp
        ../Common/StringUtils/StringUtils.cpp
        ../Common/DateTimeUtils/DateTimeUtils.cpp
        ChessEngineProjectMain.cpp
